/* Background particles */
const canvas = document.getElementById('bg');
const ctx = canvas.getContext('2d');
let W, H, particles = [];
function resize() { canvas.width = W = window.innerWidth; canvas.height = H = window.innerHeight; }
function mkParticle(init) {
  const r = Math.random();
  return { x:Math.random()*W, y:init?Math.random()*H:-5, rad:0.3+r*r*2.8, op:0.02+r*0.13, sp:0.06+r*0.35, dx:(Math.random()-.5)*0.25, wb:Math.random()*Math.PI*2 };
}
function init() { particles=[]; for(let i=0;i<160;i++) particles.push(mkParticle(true)); }
function draw() {
  ctx.clearRect(0,0,W,H);
  particles.forEach(f => {
    f.y+=f.sp; f.wb+=0.007; f.x+=Math.sin(f.wb)*f.dx;
    if(f.y>H) Object.assign(f,mkParticle(false));
    ctx.beginPath(); ctx.arc(f.x,f.y,f.rad,0,Math.PI*2);
    ctx.fillStyle=`rgba(255,255,255,${f.op})`; ctx.fill();
  });
  requestAnimationFrame(draw);
}
window.addEventListener('resize',()=>{resize();init();});
resize(); init(); draw();
