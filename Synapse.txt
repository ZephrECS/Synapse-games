const canvas = document.getElementById('bg-fx');
const ctx    = canvas.getContext('2d');
let flakes   = [];
function initSnow() {
  canvas.width = window.innerWidth; canvas.height = window.innerHeight;
  flakes = Array.from({length:140}, () => { const r=Math.random(); return {x:Math.random()*canvas.width,y:Math.random()*canvas.height,rad:.4+r*r*2.6,op:.02+r*.14,sp:.08+r*.32,dr:(Math.random()-.5)*.3,wb:Math.random()*Math.PI*2}; });
}
function drawSnow() {
  ctx.clearRect(0,0,canvas.width,canvas.height);
  flakes.forEach(f => { f.y+=f.sp; f.wb+=.007; f.x+=Math.sin(f.wb)*f.dr; if(f.y>canvas.height){f.y=-4;f.x=Math.random()*canvas.width;} ctx.beginPath(); ctx.arc(f.x,f.y,f.rad,0,Math.PI*2); ctx.fillStyle=`rgba(255,255,255,${f.op})`; ctx.fill(); });
  requestAnimationFrame(drawSnow);
}
window.addEventListener('resize', initSnow);
initSnow(); drawSnow();

  crimson:  {name:'Crimson',  color:'#ef4444','--accent':'#ef4444','--accent2':'#fca5a5','--accent3':'#fecaca'},
  sunset:   {name:'Sunset',   color:'#f97316','--accent':'#f97316','--accent2':'#fdba74','--accent3':'#fed7aa'},

  mono:     {name:'Mono',     color:'#9ca3af','--accent':'#9ca3af','--accent2':'#e5e7eb','--accent3':'#f9fafb'},

function applyTheme(key) {
  const t = THEMES[key]||THEMES.default;
  Object.entries(t).forEach(([p,v]) => { if(p.startsWith('--')) document.documentElement.style.setProperty(p,v); });
  state.theme=key; persist(); renderThemeGrid(); renderObThemeGrid();
}

/* ════════════════════════════════════════════════════
   GAMES — loaded from games.js (GAMES array)
   No API calls, no rate limits, instant load.
════════════════════════════════════════════════════ */
let activeCat     = 'all';
let activeGameIdx = null;
let isFullscreen  = false;

function initGames() {
  // GAMES is defined in games.js, loaded via <script> tag
  if (typeof GAMES === 'undefined' || !GAMES.length) {
    document.getElementById('gamesGrid').innerHTML =
      '<div style="grid-column:1/-1;color:var(--muted);text-align:center;padding:60px 0;font-size:13px;">No games found — check games.js</div>';
    return;
  }
  const cats = ['all', ...new Set(GAMES.map(g => g.cat).filter(c => c && c !== 'all'))];
  renderCategoryChips(cats);
  renderGames();
  document.getElementById('gamesPanelSub').textContent = GAMES.length + ' games';
  document.getElementById('infoGames').textContent = GAMES.length + ' games';
}

function renderCategoryChips(cats) {
  document.getElementById('filterRow').innerHTML = cats.map(c =>
    `<button class="chip ${activeCat===c?'active':''}" onclick="filterCat('${c}')">${c==='all'?'All':c.charAt(0).toUpperCase()+c.slice(1)}</button>`
  ).join('');
}

function isInLibrary(idx) { return state.library.some(l => l.idx === idx); }

function renderGames() {
  const list = activeCat === 'all' ? GAMES : GAMES.filter(g => g.cat === activeCat);
  if (!list.length) {
    document.getElementById('gamesGrid').innerHTML = '<div style="grid-column:1/-1;color:var(--muted);text-align:center;padding:60px 0;font-size:13px;">No games in this category.</div>';
    return;
  }
  document.getElementById('gamesGrid').innerHTML = list.map(g => {
    const idx   = GAMES.indexOf(g);
    const saved = isInLibrary(idx);
    return `
      <div class="game-card" data-idx="${idx}">
        <div class="gc-bg">
          ${g.image
            ? `<img class="gc-img" src="${g.image}" alt="${g.title}" onerror="this.parentElement.innerHTML='<div class=gc-placeholder>🎮</div>'">`
            : `<div class="gc-placeholder">🎮</div>`}
        </div>
        <div class="gc-label">
          ${g.tag ? `<div class="gc-tag">${g.tag}</div>` : ''}
          <div>${g.title}</div>
        </div>
        <button class="gc-save-btn ${saved?'saved':''}" title="${saved?'Saved!':'Save'}" onclick="event.stopPropagation();cardToggleLib(${idx},this)">
          ${saved
            ? `<svg viewBox="0 0 24 24" fill="currentColor" stroke="none"><path d="M20.84 4.61a5.5 5.5 0 00-7.78 0L12 5.67l-1.06-1.06a5.5 5.5 0 00-7.78 7.78l1.06 1.06L12 21.23l7.78-7.78 1.06-1.06a5.5 5.5 0 000-7.78z"/></svg>`
            : `<svg viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2" stroke-linecap="round"><path d="M20.84 4.61a5.5 5.5 0 00-7.78 0L12 5.67l-1.06-1.06a5.5 5.5 0 00-7.78 7.78l1.06 1.06L12 21.23l7.78-7.78 1.06-1.06a5.5 5.5 0 000-7.78z"/></svg>`}
        </button>
      </div>`;
  }).join('');
  document.querySelectorAll('.game-card').forEach(c => {
    c.addEventListener('click', () => clickGame(parseInt(c.dataset.idx)));
  });
}

function filterCat(cat) {
  activeCat = cat;
  document.querySelectorAll('#filterRow .chip').forEach(c =>
    c.classList.toggle('active', c.textContent.toLowerCase()===cat||(cat==='all'&&c.textContent==='All'))
  );
  renderGames();
}

function cardToggleLib(idx, btn) {
  if (isInLibrary(idx)) {
    state.library = state.library.filter(l => l.idx !== idx);
    btn.className = 'gc-save-btn'; btn.title = 'Save';
    btn.innerHTML = `<svg viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2" stroke-linecap="round"><path d="M20.84 4.61a5.5 5.5 0 00-7.78 0L12 5.67l-1.06-1.06a5.5 5.5 0 00-7.78 7.78l1.06 1.06L12 21.23l7.78-7.78 1.06-1.06a5.5 5.5 0 000-7.78z"/></svg>`;
    showToast('Removed from library');
  } else {
    const g = GAMES[idx];
    state.library.push({idx, title:g.title, image:g.image});
    btn.className = 'gc-save-btn saved ripple'; btn.title = 'Saved!';
    btn.innerHTML = `<svg viewBox="0 0 24 24" fill="currentColor" stroke="none"><path d="M20.84 4.61a5.5 5.5 0 00-7.78 0L12 5.67l-1.06-1.06a5.5 5.5 0 00-7.78 7.78l1.06 1.06L12 21.23l7.78-7.78 1.06-1.06a5.5 5.5 0 000-7.78z"/></svg>`;
    setTimeout(() => btn.classList.remove('ripple'), 600);
    showToast(`♥ ${g.title} saved!`);
  }
  persist(); renderLibraryPanel(); renderLibrary();
}

function clickGame(idx) {
  const g = GAMES[idx];
  if (g) launchGame(g.url, g.title);
}

/* ════════════════════════════════════════════════════
   LAUNCH GAME
════════════════════════════════════════════════════ */
async function launchGame(url, title) {
  const ov = document.createElement('div');
  ov.className = 'game-launch-ov';
  ov.innerHTML = `<div class="launch-spinner-ring"></div><div class="launch-title">${title||'Loading...'}</div><div class="launch-sub">Preparing game</div>`;
  document.body.appendChild(ov);
  try {
    const res  = await fetch(url);
    const code = await res.text();
    await new Promise(r => setTimeout(r, 600));
    ov.classList.add('out'); setTimeout(() => ov.remove(), 280);
    const wrap   = document.getElementById('game-frame-wrap');
    const iframe = document.getElementById('game-iframe');
    wrap.classList.add('active');
    iframe.contentDocument.open(); iframe.contentDocument.write(code); iframe.contentDocument.close();
    showGameControls();
  } catch(e) {
    ov.classList.add('out'); setTimeout(() => ov.remove(), 280);
    showToast('⚠ Failed to load game');
  }
}

/* ════════════════════════════════════════════════════

function showGameControls() {
  const hud=document.getElementById('hud'), ctrl=document.getElementById('hudGameControls');
  ctrl.style.display='flex';
  requestAnimationFrame(()=>requestAnimationFrame(()=>{ hud.classList.add('game-mode'); ctrl.classList.add('visible'); }));
}
function hideGameControls() {
  const hud=document.getElementById('hud'), ctrl=document.getElementById('hudGameControls');
  hud.classList.remove('game-mode'); ctrl.classList.remove('visible');
  setTimeout(()=>{ ctrl.style.display='none'; }, 420);
}
function gcToggleFullscreen() {
  const wrap=document.getElementById('game-frame-wrap');
  const icon=document.getElementById('gcFsIcon'), txt=document.getElementById('gcFsTxt');
  if (!isFullscreen) {
    wrap.requestFullscreen?.() || wrap.webkitRequestFullscreen?.();
    isFullscreen=true; icon.innerHTML=`<path d="M8 3v3a2 2 0 01-2 2H3m18 0h-3a2 2 0 01-2-2V3m0 18v-3a2 2 0 012-2h3M3 16h3a2 2 0 012 2v3"/>`; txt.textContent='Exit Full';
  } else {
    document.exitFullscreen?.() || document.webkitExitFullscreen?.();
    isFullscreen=false; icon.innerHTML=`<path d="M8 3H5a2 2 0 00-2 2v3m18 0V5a2 2 0 00-2-2h-3m0 18h3a2 2 0 002-2v-3M3 16v3a2 2 0 002 2h3"/>`; txt.textContent='Fullscreen';
  }
}
document.addEventListener('fullscreenchange', () => {
  if (!document.fullscreenElement && isFullscreen) {
    isFullscreen=false;
    document.getElementById('gcFsIcon').innerHTML=`<path d="M8 3H5a2 2 0 00-2 2v3m18 0V5a2 2 0 00-2-2h-3m0 18h3a2 2 0 002-2v-3M3 16v3a2 2 0 002 2h3"/>`;
    document.getElementById('gcFsTxt').textContent='Fullscreen';
  }
});
function exitGame() {
  const wrap=document.getElementById('game-frame-wrap'), iframe=document.getElementById('game-iframe');
  if (isFullscreen){document.exitFullscreen?.(); isFullscreen=false;}
  wrap.style.transition='opacity .28s ease'; wrap.style.opacity='0';
  setTimeout(()=>{ wrap.classList.remove('active'); wrap.style.opacity=''; wrap.style.transition=''; iframe.src='about:blank'; hideGameControls(); }, 280);
}

════════════════════════════════════════════════════ */
function renderLibraryPanel() {
  const grid=document.getElementById('libPanelGrid'); if(!grid)return;
  if (!state.library?.length) {
    grid.innerHTML=`<div class="lib-empty-state"><svg viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="1.3"><path d="M20.84 4.61a5.5 5.5 0 00-7.78 0L12 5.67l-1.06-1.06a5.5 5.5 0 00-7.78 7.78l1.06 1.06L12 21.23l7.78-7.78 1.06-1.06a5.5 5.5 0 000-7.78z"/></svg><p>No games saved yet.<br>Hover a game card and click the heart.</p></div>`;
    return;
  }
  grid.innerHTML = state.library.map((l,i) => `
    <div class="lib-panel-card">
      ${l.image?`<img src="${l.image}" alt="${l.title}" onerror="this.style.display='none'" onclick="clickGame(${l.idx})">` : ''}
      <div class="lib-panel-card-foot">
        <div class="lib-panel-card-title" onclick="clickGame(${l.idx})">${l.title}</div>
        <button class="lib-panel-remove" onclick="removeFromLib(${i})">✕</button>
      </div>
    </div>`).join('');
}
function removeFromLib(i) {
  state.library.splice(i,1); persist();
  renderLibraryPanel(); renderLibrary(); renderGames();
  showToast('Removed from library');
}

  else{form.style.display='';logged.classList.remove('show');avatar.textContent='?';}

async function exportData() {
  const ls={};
  for(let i=0;i<localStorage.length;i++){const k=localStorage.key(i);ls[k]=localStorage.getItem(k);}
  const blob=new Blob([JSON.stringify({synapseState:state,localStorage:ls,exportedAt:new Date().toISOString()},null,2)],{type:'application/json'});
  const a=document.createElement('a');a.href=URL.createObjectURL(blob);a.download='synapse_data.json';a.click();
  setTimeout(()=>URL.revokeObjectURL(a.href),1000);
  showToast('✓ Exported!');
}

function switchTab(tab) {
  document.querySelectorAll('.hud-btn[data-tab],.hud-icon-btn[data-tab]').forEach(b=>b.classList.toggle('active',b.dataset.tab===tab));
  document.querySelectorAll('.panel').forEach(p=>p.classList.toggle('active',p.id==='panel-'+tab));
  if(tab==='settings')renderThemeGrid();
  if(tab==='account'){renderAccount();renderLibrary();}
  if(tab==='library')renderLibraryPanel();
}
document.querySelectorAll('.hud-btn[data-tab],.hud-icon-btn[data-tab]').forEach(b=>{
  b.addEventListener('click',()=>switchTab(b.dataset.tab));
});

function proxyKey(e){if(e.key==='Enter')goProxy();}
function goProxy(){
  let url=document.getElementById('proxyInput').value.trim(); if(!url)return;
  if(!url.startsWith('http'))url='https://'+url;
  document.getElementById('proxyWrap').innerHTML=`<iframe src="${url}" style="width:100%;height:100%;border:none;" sandbox="allow-scripts allow-same-origin allow-forms"></iframe>`;
}
function homeSearchKey(e){
  if(e.key!=='Enter')return;
  const v=document.getElementById('homeSearch').value.trim(); if(!v)return;
  const url=v.includes('.')&&!v.includes(' ')?(v.startsWith('http')?v:'https://'+v):`https://www.google.com/search?q=${encodeURIComponent(v)}`;
  if(state.experience==='full'){switchTab('proxy');document.getElementById('proxyInput').value=url;goProxy();}
  else window.open(url,'_blank');
}
