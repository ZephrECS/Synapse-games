function makeBgParticle(init=false){
  const r = Math.random();
  const cx = currentBg;
  if(cx===0) return {x:Math.random()*bgW,y:init?Math.random()*bgH:-5,rad:.3+r*r*2.8,op:.02+r*.13,sp:.06+r*.35,dx:(Math.random()-.5)*.25,wb:Math.random()*Math.PI*2};
  if(cx===1) return {x:Math.floor(Math.random()*(bgW/14))*14,y:init?Math.random()*bgH:-20,sp:2+r*8,len:Math.floor(2+r*18),op:0.04+r*0.18,chars:'0123456789ABCDEF'.split(''),cur:Math.floor(Math.random()*16)};
  if(cx===2) return {x:Math.random()*bgW,y:Math.random()*bgH,r:1+r*4,op:.02+r*.12,vx:(Math.random()-.5)*.3,vy:(Math.random()-.5)*.3,hue:220+r*120};

  if(cx===4) return {x:Math.random()*bgW,y:Math.random()*bgH,r:.5+r*2.5,op:0,maxOp:.04+r*.18,phase:Math.random()*Math.PI*2,spd:.01+r*.03,vx:(Math.random()-.5)*.4,vy:(Math.random()-.5)*.4};

/* ═══════════════════════════════════════════════════════
   GAME DATA — per-game localStorage stats
   Key format: syn_gd_<username|"guest">_<gameIdx>
   Each record: { playCount, totalSecs, lastPlayed }
═══════════════════════════════════════════════════════ */
function gdKey(idx){ return `syn_gd_${state.username||'guest'}_${idx}`; }
function gdGet(idx){ try{ const r=localStorage.getItem(gdKey(idx)); return r?JSON.parse(r):null; }catch(e){return null;} }
function gdSet(idx,data){ try{ localStorage.setItem(gdKey(idx),JSON.stringify(data)); }catch(e){} }

function gdFormatTime(secs){
  if(!secs||secs<60) return secs?`${secs}s`:'—';
  const h=Math.floor(secs/3600), m=Math.floor((secs%3600)/60);
  return h>0?`${h}h ${m}m`:`${m}m`;
}
function gdFormatDate(iso){
  if(!iso) return '';
  const d=new Date(iso), now=new Date();
  const diff=Math.floor((now-d)/864e5);
  if(diff===0) return 'today';
  if(diff===1) return 'yesterday';
  if(diff<7) return `${diff}d ago`;
  return d.toLocaleDateString(undefined,{month:'short',day:'numeric'});
}

// Session tracking
let _gdActiveIdx = null;
let _gdSessionStart = null;

function gdStartSession(idx){
  _gdActiveIdx = idx;
  _gdSessionStart = Date.now();
}
function gdEndSession(){
  if(_gdActiveIdx===null || !_gdSessionStart) return;
  const secs = Math.floor((Date.now()-_gdSessionStart)/1000);
  if(secs<2){ _gdActiveIdx=null;_gdSessionStart=null; return; } // ignore accidental opens
  const existing = gdGet(_gdActiveIdx)||{playCount:0,totalSecs:0,lastPlayed:null};
  existing.playCount = (existing.playCount||0)+1;
  existing.totalSecs = (existing.totalSecs||0)+secs;
  existing.lastPlayed = new Date().toISOString();
  gdSet(_gdActiveIdx, existing);
  _gdActiveIdx=null; _gdSessionStart=null;
}

// All game stats for the current user
function gdAllStats(){
  const out=[];
  GAMES.forEach((g,i)=>{
    const d=gdGet(i);
    if(d&&d.playCount>0) out.push({idx:i,game:g,...d});
  });
  return out.sort((a,b)=>new Date(b.lastPlayed)-new Date(a.lastPlayed));
}

/* Build the localStorage shim script for a given game index.
   All reads/writes inside the iframe get transparently redirected
   to Synapse's localStorage under namespaced keys:
   syn_gsave_<user|guest>_<idx>_<originalKey>
   The shim also patches sessionStorage the same way so games
   that use it don't silently lose data either. */
function buildStorageShim(idx){
  const ns = `syn_gsave_${state.username||'guest'}_${idx}_`;
  return `<script>
(function(){
  var NS = ${JSON.stringify(ns)};
  var _raw = window.localStorage || {};

  function shimStorage(target){
    var store = {
      getItem: function(k){ return localStorage.getItem(NS+k); },
      setItem: function(k,v){ try{ localStorage.setItem(NS+k,String(v)); }catch(e){} },
      removeItem: function(k){ localStorage.removeItem(NS+k); },
      clear: function(){
        Object.keys(localStorage).filter(function(k){ return k.indexOf(NS)===0; })
          .forEach(function(k){ localStorage.removeItem(k); });
      },
      key: function(n){
        var keys=Object.keys(localStorage).filter(function(k){ return k.indexOf(NS)===0; });
        return keys[n]?keys[n].slice(NS.length):null;
      },
      get length(){
        return Object.keys(localStorage).filter(function(k){ return k.indexOf(NS)===0; }).length;
      }
    };
    // Proxy property access (game[ls] = value style)
    try{
      return new Proxy(store,{
        get: function(t,p){
          if(p in t) return typeof t[p]==='function'?t[p].bind(t):t[p];
          return localStorage.getItem(NS+p);
        },
        set: function(t,p,v){
          if(p in t){ t[p]=v; return true; }
          try{ localStorage.setItem(NS+p,String(v)); }catch(e){}
          return true;
        },
        deleteProperty: function(t,p){
          localStorage.removeItem(NS+p); return true;
        }
      });
    }catch(e){ return store; }
  }

  try{ Object.defineProperty(window,'localStorage',{get:function(){ return shimStorage(); },configurable:true}); }catch(e){}
  try{ Object.defineProperty(window,'sessionStorage',{get:function(){ return shimStorage(); },configurable:true}); }catch(e){}
})();
<\/script>`;
}

async function launchGame(idx,url,title){
  const ov=document.createElement('div');ov.className='game-launch-ov';
  ov.innerHTML=`<div class="launch-spinner-ring"></div><div class="launch-title">${title||'Loading...'}</div><div class="launch-sub">Preparing game</div>`;
  document.body.appendChild(ov);
  try{
    const res=await fetch(url);let code=await res.text();
    const trimmed=code.trim();
    const isBareUrl=/^https?:\/\/\S+$/.test(trimmed);
    await new Promise(r=>setTimeout(r,600));
    ov.classList.add('out');setTimeout(()=>ov.remove(),280);
    const wrap=document.getElementById('game-frame-wrap'),iframe=document.getElementById('game-iframe');
    wrap.classList.add('active');

    if(url.endsWith('.txt')||isBareUrl){
      // Bare URL game — load it inside a thin wrapper page that injects the shim
      // via postMessage + eval before the inner page loads. Since we can't inject
      // into a cross-origin page directly, we load it inside a nested iframe inside
      // our shim page (same-origin blob), which overrides localStorage on the outer
      // document that the game sees when loaded via blob:.
      //
      // For these games, just load directly — their origin persists naturally.
      iframe.src=trimmed;
    } else {
      // Inline HTML game — inject shim as very first thing in <head> or <body>
      const shim = buildStorageShim(idx);
      // Try to inject after <head> open tag, fall back to prepend before everything
      if(/<head[\s>]/i.test(code)){
        code = code.replace(/(<head[^>]*>)/i, '$1\n'+shim);
      } else if(/<html[\s>]/i.test(code)){
        code = code.replace(/(<html[^>]*>)/i, '$1\n'+shim);
      } else {
        code = shim + '\n' + code;
      }
      iframe.contentDocument.open();
      iframe.contentDocument.write(code);
      iframe.contentDocument.close();
    }
    gdStartSession(idx);
    showGameControls();
  }catch(e){ov.classList.add('out');setTimeout(()=>ov.remove(),280);showToast('⚠ Failed to load game');}
}

function gcToggleFullscreen(){
  const iframe=document.getElementById('game-iframe');
  const icon=document.getElementById('gcFsIcon'),txt=document.getElementById('gcFsTxt');
  if(!isFullscreen){
    // Request fullscreen on the iframe itself so the game gets true full viewport
    (iframe.requestFullscreen||iframe.webkitRequestFullscreen||iframe.mozRequestFullScreen||iframe.msRequestFullscreen).call(iframe);
    isFullscreen=true;
    icon.innerHTML='<path d="M8 3v3a2 2 0 01-2 2H3m18 0h-3a2 2 0 01-2-2V3m0 18v-3a2 2 0 012-2h3M3 16h3a2 2 0 012 2v3"/>';
    txt.textContent='Exit Full';
  } else {
    (document.exitFullscreen||document.webkitExitFullscreen||document.mozCancelFullScreen||document.msExitFullscreen).call(document);
    isFullscreen=false;
    icon.innerHTML='<path d="M8 3H5a2 2 0 00-2 2v3m18 0V5a2 2 0 00-2-2h-3m0 18h3a2 2 0 002-2v-3M3 16v3a2 2 0 002 2h3"/>';
    txt.textContent='Fullscreen';
  }
}

  if(tv) tv.textContent=gdFormatTime(totalSecs)||'0m';

  const recent=document.getElementById('acctRecentGames');
  if(!recent) return;
