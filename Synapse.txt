// Rewrite Google Apps Script macro URLs to raw jsdelivr CDN equivalents
// e.g. https://script.google.com/macros/s/.../exec?game=342.html
//   -> https://cdn.jsdelivr.net/gh/NoahsAmazingTutoringHelp/Noahs-Calculus-Tutor@master/games/342.html
const GAME_URL_OVERRIDES = {
  'cloverpit': 'https://raw.githubusercontent.com/ZephrECS/Synapse-games/refs/heads/main/CloverPit/Clover%20Pit.html',
};
function rewriteGameUrl(g){
  if(!g.url) return g;
  // Check title-based overrides first
  const titleKey = (g.title||'').toLowerCase().replace(/[^a-z0-9]/g,'');
  if(GAME_URL_OVERRIDES[titleKey]) return {...g, url: GAME_URL_OVERRIDES[titleKey]};
  try{
    const u=new URL(g.url);
    if(u.hostname==='script.google.com' && u.pathname.includes('/macros/')){
      const game=u.searchParams.get('game');
      if(game){
        return {...g, url:`https://cdn.jsdelivr.net/gh/NoahsAmazingTutoringHelp/Noahs-Calculus-Tutor@master/games/${game}`};
      }
    }
  }catch(e){}
  return g;
}

/* NEW TAB LAUNCHER */
async function launchGameNewTab(idx,url,title){
  const ov=document.createElement('div');
  ov.className='game-launch-ov';
  ov.innerHTML=`<div class="launch-spinner-ring"></div><div class="launch-title">${title||'Loading...'}</div><div class="launch-sub">Opening in new tab</div>`;
  document.body.appendChild(ov);
  try{
    const res=await fetch(url);
    let code=await res.text();
    const trimmed=code.trim();
    const isBareUrl=/^https?:\/\/\S+$/.test(trimmed);
    await new Promise(r=>setTimeout(r,480));
    ov.classList.add('out');setTimeout(()=>ov.remove(),280);
    // Record that we opened this game (for playtime tracking)
    try{localStorage.setItem('syn_newtab_open',JSON.stringify({idx,opened:Date.now()}));}catch(e){}
    if(isBareUrl){
      const t=window.open(trimmed,'_blank');
      if(!t)showToast('&#9888; Pop-up blocked — please allow pop-ups for this site.');
    }else{
      const shim=buildStorageShim(idx);
      if(/<head[\s>]/i.test(code)){code=code.replace(/(<head[^>]*>)/i,'$1\n'+shim);}
      else if(/<html[\s>]/i.test(code)){code=code.replace(/(<html[^>]*>)/i,'$1\n'+shim);}
      else{code=shim+'\n'+code;}
      if(!/<html[\s>]/i.test(code)){code=`<!DOCTYPE html><html><head><meta charset="UTF-8"><title>${title}</title><style>*{margin:0;padding:0;}body{background:#000;}</style></head><body>${code}</body></html>`;}
      const t=window.open('about:blank','_blank');
      if(!t){showToast('&#9888; Pop-up blocked — please allow pop-ups for this site.');}
      else{t.document.open();t.document.write(code);t.document.close();}
    }
  }catch(e){
    // Fetch failed (likely CORS) — fall back to opening the URL directly
    ov.classList.add('out');setTimeout(()=>ov.remove(),280);
    try{localStorage.setItem('syn_newtab_open',JSON.stringify({idx,opened:Date.now()}));}catch(e2){}
    const t=window.open(url,'_blank');
    if(!t)showToast('&#9888; Pop-up blocked — please allow pop-ups for this site.');
  }
}

function buildStorageShim(idx){
  const ns=`syn_gsave_${state.username||'guest'}_${idx}_`;
  return `<script>(function(){var NS=${JSON.stringify(ns)};function shimStorage(){var store={getItem:function(k){return localStorage.getItem(NS+k);},setItem:function(k,v){try{localStorage.setItem(NS+k,String(v));}catch(e){}},removeItem:function(k){localStorage.removeItem(NS+k);},clear:function(){Object.keys(localStorage).filter(function(k){return k.indexOf(NS)===0;}).forEach(function(k){localStorage.removeItem(k);});},key:function(n){var keys=Object.keys(localStorage).filter(function(k){return k.indexOf(NS)===0;});return keys[n]?keys[n].slice(NS.length):null;},get length(){return Object.keys(localStorage).filter(function(k){return k.indexOf(NS)===0;}).length;}};try{return new Proxy(store,{get:function(t,p){if(p in t)return typeof t[p]==='function'?t[p].bind(t):t[p];return localStorage.getItem(NS+p);},set:function(t,p,v){if(p in t){t[p]=v;return true;}try{localStorage.setItem(NS+p,String(v));}catch(e){}return true;},deleteProperty:function(t,p){localStorage.removeItem(NS+p);return true;}});}catch(e){return store;}}try{Object.defineProperty(window,'localStorage',{get:function(){return shimStorage();},configurable:true});}catch(e){}try{Object.defineProperty(window,'sessionStorage',{get:function(){return shimStorage();},configurable:true});}catch(e){}})();<\/script>`;
}

/* Track playtime: when user returns to this tab after opening a game */
document.addEventListener('visibilitychange',()=>{
  if(document.visibilityState==='visible'){
    try{
      const raw=localStorage.getItem('syn_newtab_open');
      if(!raw)return;
      const sess=JSON.parse(raw);
      const secs=Math.floor((Date.now()-sess.opened)/1000);
      if(secs>5){
        const existing=gdGet(sess.idx)||{playCount:0,totalSecs:0,lastPlayed:null};
        existing.playCount=(existing.playCount||0)+1;
        existing.totalSecs=(existing.totalSecs||0)+secs;
        existing.lastPlayed=new Date().toISOString();
        gdSet(sess.idx,existing);
        localStorage.removeItem('syn_newtab_open');
        renderGames();renderAccountStats();
      }
    }catch(e){}
  }
});
